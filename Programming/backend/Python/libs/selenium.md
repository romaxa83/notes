    if not agree_checkbox.is_selected():

    # –û—Ç–ø—Ä–∞–≤–∫–∞ —Ñ–æ—Ä–º—ã
    driver.find_element(By.ID, "submit").click()
    
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º —É—Å–ø–µ—à–Ω—É—é –æ—Ç–ø—Ä–∞–≤–∫—É
    success_msg = wait.until(

    )

finally:
    driver.quit()
