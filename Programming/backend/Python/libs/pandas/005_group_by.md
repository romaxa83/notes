#### üîπ –°–æ–∑–¥–∞–Ω–∏–µ –æ–±—ä–µ–∫—Ç–∞ GROUPBY

```python
food_data = {
	"Item": ["Banana", "Cucumber", "Orange", "Tomato", "Watermelon"],
	"Type": ["Fruit", "Vegetable", "Fruit", "Vegetable", "Fruit"],
	"Price": [0.99, 1.25, 0.25, 0.33, 3.00]
}
supermarket = pd.DataFrame(data = food_data)
# Out
#   Item  Type   Price
# -------------------------------
# 0 Banana Fruit 0.99
# 1 Cucumber Vegetable 1.25
# 2 Orange Fruit 0.25
# 3 Tomato Vegetable 0.33
# 4 Watermelon Fruit 3.00

# –º–µ—Ç–æ–¥—É groupby –æ–±—ä–µ–∫—Ç–∞ DataFrame –Ω—É–∂–Ω–æ –ø–µ—Ä–µ–¥–∞—Ç—å —Å—Ç–æ–ª–±–µ—Ü, –Ω–∞ –æ—Å–Ω–æ–≤–µ –∫–æ—Ç–æ—Ä–æ–≥–æ pandas –≤—ã–ø–æ–ª–Ω–∏—Ç –≥—Ä—É–ø–ø–∏—Ä–æ–≤–∫—É, –º–µ—Ç–æ–¥ –≤–µ—Ä–Ω–µ—Ç –æ–±—ä–µ–∫—Ç DataFrameGroupBy
groups = supermarket.groupby("Type")
groups.get_group("Fruit")
# Out
#   Item   Type   Price
# ------------------------
# 0 Banana Fruit 0.99
# 2 Orange Fruit 0.25
# 4 Watermelon Fruit 3.00

# –≤—ã—á–∏—Å–ª—è–µ–º —Å—Ä–µ–¥–Ω—é—é —Ü–µ–Ω—É —Ñ—Ä—É–∫—Ç–æ–≤ –∏ –æ–≤–æ—â–µ–π –≤ —Å—É–ø–µ—Ä–º–∞—Ä–∫–µ—Ç–µ.
groups.mean()
# Out
# Type Price
# ----------------
# Fruit 1.413333
# Vegetable 0.790000
```

#### üîπ –ê–≥—Ä–µ–≥–∞—Ç–Ω—ã–µ –æ–ø–µ—Ä–∞—Ü–∏–∏

```python
# –ú–µ—Ç–æ–¥ sum, —Å–∫–ª–∞–¥—ã–≤–∞–µ—Ç –∑–Ω–∞—á–µ–Ω–∏—è —Å—Ç–æ–ª–±—Ü–æ–≤ –≤ –∫–∞–∂–¥–æ–π –≥—Ä—É–ø–ø–µ. –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é pandas —Ä–∞—Å—Å—á–∏—Ç—ã–≤–∞—Ç—å —Å—É–º–º—ã –≤–æ –≤—Å–µ—Ö —á–∏—Å–ª–æ–≤—ã—Ö —Å—Ç–æ–ª–±—Ü–∞—Ö –≤ DataFrame.

# Out

# --------------------------------------------

# –ü—Ä–æ—Å—á–µ—Ç –ø–æ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–º—É —Å—Ç–æ–ª–±—Ü—É
sectors["Revenues"].sum()

# –º–µ—Ç–æ–¥ mean, –≤—ã—á–∏—Å–ª—è—é—â–∏–π —Å—Ä–µ–¥–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è –ø–æ —Å—Ç–æ–ª–±—Ü–∞–º
sectors.mean()

# –º–µ—Ç–æ–¥ max, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–∞–Ω—á–µ–Ω–∏–µ –≤ –≥—Ä—É–ø–ø–∞—Ö
sectors["Profits"].max()

#¬†–º–µ—Ç–æ–¥ min –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –≤ –∑–∞–¥–∞–Ω–Ω–æ–º —Å—Ç–æ–ª–±—Ü–µ –ø–æ –≥—Ä—É–ø–ø–∞–º
sectors["Profits"].min()

# –ú–µ—Ç–æ–¥ agg –ø—Ä–∏–º–µ–Ω—è–µ—Ç –Ω–µ—Å–∫–æ–ª—å–∫–æ –∞–≥—Ä–µ–≥–∞—Ç–Ω—ã—Ö –æ–ø–µ—Ä–∞—Ü–∏–π –∫ —Ä–∞–∑–Ω—ã–º —Å—Ç–æ–ª–±—Ü–∞–º –∏ –ø—Ä–∏–Ω–∏–º–∞–µ—Ç —Å–ª–æ–≤–∞—Ä—å –≤ –∫–∞—á–µ—Å—Ç–≤–µ –∞—Ä–≥—É–º–µ–Ω—Ç–∞.
aggregations = {
	"Revenues": "min",
	"Profits": "max",
	"Employees": "mean"
}
sectors.agg(aggregations).head()
```

#### üîπ –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —Å–æ–±—Å—Ç–≤–µ–Ω–Ω—ã—Ö –æ–ø–µ—Ä–∞—Ü–∏–π –∫–æ –≤—Å–µ–º –≥—Ä—É–ø–ø–∞–º –Ω–∞–±–æ—Ä–∞

```python
# –ú–µ—Ç–æ–¥ nlargest –æ–±—ä–µ–∫—Ç–∞ DataFrame –∏–∑–≤–ª–µ–∫–∞–µ—Ç –∑–∞–ø–∏—Å–∏ —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º –∑–Ω–∞—á–µ–Ω–∏–µ–º –≤ –∑–∞–¥–∞–Ω–Ω–æ–º —Å—Ç–æ–ª–±—Ü–µ
fortune.nlargest(n = 5, columns = "Profits")

# –º–µ—Ç–æ–¥ apply –æ–±—ä–µ–∫—Ç–∞ GroupBy –ø—Ä–∏–Ω–∏–º–∞–µ—Ç —Ñ—É–Ω–∫—Ü–∏—é, –≤—ã–∑—ã–≤–∞–µ—Ç –µ–µ –æ–¥–∏–Ω —Ä–∞–∑ –¥–ª—è –∫–∞–∂–¥–æ–π –≥—Ä—É–ø–ø—ã –≤ –æ–±—ä–µ–∫—Ç–µ GroupBy, —Å–æ–±–∏—Ä–∞–µ—Ç –ø–æ–ª—É—á–µ–Ω–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏—Ö –≤ –Ω–æ–≤–æ–º DataFrame

# —Ñ—É–Ω–∫—Ü–∏—è –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –æ–¥–∏–Ω –∞—Ä–≥—É–º–µ–Ω—Ç: DataFrame –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∑–∞–ø–∏—Å—å –∏–∑ DataFrame —Å –Ω–∞–∏–±–æ–ª—å—à–∏–º –∑–Ω–∞—á–µ–Ω–∏–µ–º –≤ —Å—Ç–æ–ª–±—Ü–µ Revenues
def get_largest_row(df):
	return df.nlargest(1, "Revenues")

# get_largest_row –≤—ã–∑–æ–≤–∏—Ç—Å—è –æ–¥–∏–Ω —Ä–∞–∑ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–µ–∫—Ç–æ—Ä–∞ –∏ –≤–µ—Ä–Ω–µ—Ç DataFrame
sectors.apply(get_largest_row).head()
```

#### üîπ –ì—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∞ –ø–æ –Ω–µ—Å–∫–æ–ª—å–∫–∏–º —Å—Ç–æ–ª–±—Ü–∞–º
–ú–æ–∂–Ω–æ —Å–æ–∑–¥–∞—Ç—å –æ–±—ä–µ–∫—Ç GroupBy, —Å–≥—Ä—É–ø–ø–∏—Ä–æ–≤–∞–≤ –¥–∞–Ω–Ω—ã–µ –ø–æ –∑–Ω–∞—á–µ–Ω–∏—è–º –∏–∑ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö —Å—Ç–æ–ª–±—Ü–æ–≤ –≤ DataFrame. –≠—Ç–∞ –≤–æ–∑–º–æ–∂–Ω–æ—Å—Ç—å –æ—Å–æ–±–µ–Ω–Ω–æ —É–¥–æ–±–Ω–∞, –∫–æ–≥–¥–∞ –≥—Ä—É–ø–ø—ã –∏–¥–µ–Ω—Ç–∏—Ñ–∏—Ü–∏—Ä—É—é—Ç—Å—è –∑–Ω–∞—á–µ–Ω–∏—è–º–∏ –∫–æ–º–±–∏–Ω–∞—Ü–∏–π —Å—Ç–æ–ª–±—Ü–æ–≤.
