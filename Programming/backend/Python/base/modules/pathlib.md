#### üîπ –ü–æ–∏—Å–∫ —Ñ–∞–π–ª–æ–≤ —Å –ø–æ–º–æ—â—å—é glob –∏ rglob

–í–∞–º –Ω—É–∂–Ω–æ –Ω–∞–π—Ç–∏ –≤—Å–µ¬†`.py`¬†—Ñ–∞–π–ª—ã –≤ –ø—Ä–æ–µ–∫—Ç–µ? –ò–ª–∏ –≤—Å–µ¬†`.jpg`¬†–∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –≤ –ø–∞–ø–∫–µ —Å —Ñ–æ—Ç–æ–≥—Ä–∞—Ñ–∏—è–º–∏? –î–ª—è —ç—Ç–æ–≥–æ –±–æ–ª—å—à–µ –Ω–µ –Ω—É–∂–µ–Ω –æ—Ç–¥–µ–ª—å–Ω—ã–π –º–æ–¥—É–ª—å¬†`glob`. –ú–µ—Ç–æ–¥—ã –¥–ª—è –ø–æ–∏—Å–∫–∞ –ø–æ —à–∞–±–ª–æ–Ω—É –≤—Å—Ç—Ä–æ–µ–Ω—ã –ø—Ä—è–º–æ –≤ –æ–±—ä–µ–∫—Ç—ã¬†`Path`.
- `.glob(pattern)`¬†‚Äî –∏—â–µ—Ç —Ñ–∞–π–ª—ã –ø–æ —à–∞–±–ª–æ–Ω—É –≤ —Ç–µ–∫—É—â–µ–π –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏.
- `.rglob(pattern)`¬†‚Äî –∏—â–µ—Ç —Ñ–∞–π–ª—ã —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ (`r`¬†–æ—Ç "recursive") –≤–æ –≤—Å–µ—Ö –ø–æ–¥–¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—è—Ö.

```python

# rglob –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –≥–µ–Ω–µ—Ä–∞—Ç–æ—Ä, —á—Ç–æ —ç–∫–æ–Ω–æ–º–∏—Ç –ø–∞–º—è—Ç—å
py_files_generator = project_dir.rglob('*.py')

for path_obj in py_files_generator:
    # –ú—ã —Å—Ä–∞–∑—É –ø–æ–ª—É—á–∞–µ–º Path –æ–±—ä–µ–∫—Ç—ã, –≥–æ—Ç–æ–≤—ã–µ –∫ —Ä–∞–±–æ—Ç–µ!
    print(f"–ù–∞–π–¥–µ–Ω —Ñ–∞–π–ª: {path_obj}, —Ä–∞–∑–º–µ—Ä: {path_obj.stat().st_size} –±–∞–π—Ç")
```

#### üîπ –†–∞–±–æ—Ç–∞ —Å –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–º–∏ —Ñ–∞–π–ª–æ–≤

```python
# 2. –§–∏–ª—å—Ç—Ä—É–µ–º, –æ—Å—Ç–∞–≤–ª—è—è —Ç–æ–ª—å–∫–æ —Ñ–∞–π–ª—ã (–Ω–∞ —Å–ª—É—á–∞–π, –µ—Å–ª–∏ –ø–æ–ø–∞–¥–µ—Ç—Å—è —Å—Å—ã–ª–∫–∞ –Ω–∞ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é)
# 3. –°–æ–∑–¥–∞–µ–º —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π (—Ä–∞–∑–º–µ—Ä_—Ñ–∞–π–ª–∞, –ø—É—Ç—å_–∫_—Ñ–∞–π–ª—É)

    for p in search_dir.rglob('*')
    if p.is_file()

for size, path in all_files[:10]:
    # –ü—Ä–∏–≤–æ–¥–∏–º —Ä–∞–∑–º–µ—Ä –∫ –º–µ–≥–∞–±–∞–π—Ç–∞–º –¥–ª—è —á–∏—Ç–∞–µ–º–æ—Å—Ç–∏
    print(f"  {path.name:<40} | {size / 1024 / 1024:.2f} MB")
```

**–ó–∞–¥–∞—á–∞:**¬†–ü–µ—Ä–µ–º–µ—Å—Ç–∏—Ç—å –≤—Å–µ¬†`.jpg`¬†–∏¬†`.png`¬†–≤ –ø–∞–ø–∫—É¬†`Images`,¬†`.pdf`¬†–∏¬†`.docx`¬†‚Äî –≤¬†`Documents`,¬†`.zip`¬†–∏¬†`.rar`¬†‚Äî –≤¬†`Archives`, –∞ –≤—Å–µ –æ—Å—Ç–∞–ª—å–Ω–æ–µ –æ—Å—Ç–∞–≤–∏—Ç—å –Ω–∞ –º–µ—Å—Ç–µ.

```python
from pathlib import Path

# 1. –û–ø—Ä–µ–¥–µ–ª—è–µ–º –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é –¥–ª—è —Ä–∞–±–æ—Ç—ã
source_dir = Path.home() / 'Downloads'

# 2. –û–ø—Ä–µ–¥–µ–ª—è–µ–º "–∫–∞—Ä—Ç—É" —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è: —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ -> –ø–∞–ø–∫–∞
DEST_MAP = {
    ".jpg": "Images",
    ".jpeg": "Images",
    ".png": "Images",
    ".gif": "Images",
    ".pdf": "Documents",
    ".docx": "Documents",
    ".txt": "Documents",
    ".zip": "Archives",
    ".rar": "Archives",
}

# 3. –ò—Ç–µ—Ä–∏—Ä—É–µ–º—Å—è –ø–æ –≤—Å–µ–º —Ñ–∞–π–ª–∞–º –≤ –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏
for path in source_dir.iterdir():
    # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏ –∏ –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ–º —Ç–æ–ª—å–∫–æ —Ñ–∞–π–ª—ã
    if path.is_dir():
        continue

    # 4. –ï—Å–ª–∏ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ —Ñ–∞–π–ª–∞ –µ—Å—Ç—å –≤ –Ω–∞—à–µ–π –∫–∞—Ä—Ç–µ
    if path.suffix in DEST_MAP:
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –ø–∞–ø–∫—É –Ω–∞–∑–Ω–∞—á–µ–Ω–∏—è
        dest_dir_name = DEST_MAP[path.suffix]
        dest_dir = source_dir / dest_dir_name

        # 5. –°–æ–∑–¥–∞–µ–º –ø–∞–ø–∫—É –Ω–∞–∑–Ω–∞—á–µ–Ω–∏—è, –µ—Å–ª–∏ –µ–µ –Ω–µ—Ç
        dest_dir.mkdir(exist_ok=True)

        # 6. –ü–µ—Ä–µ–º–µ—â–∞–µ–º —Ñ–∞–π–ª
        # –î–ª—è –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏—è –º–µ–∂–¥—É —Ä–∞–∑–Ω—ã–º–∏ –¥–∏—Å–∫–∞–º–∏ –ª—É—á—à–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å shutil.move
        # –ù–æ –¥–ª—è –ø—Ä–æ—Å—Ç–æ–≥–æ –ø–µ—Ä–µ–º–µ—â–µ–Ω–∏—è path.rename() –∏–¥–µ–∞–ª–µ–Ω
        print(f"–ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ {path.name} –≤ {dest_dir_name}...")
        path.rename(dest_dir / path.name)
```
