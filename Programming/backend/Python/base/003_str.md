#### üîπ F-—Å—Ç—Ä–æ–∫–∏

F-—Å—Ç—Ä–æ–∫–∏ (formatted string literals) - —ç—Ç–æ —Å–æ–≤—Ä–µ–º–µ–Ω–Ω—ã–π –∏ –º–æ—â–Ω—ã–π —Å–ø–æ—Å–æ–± —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è —Å—Ç—Ä–æ–∫ –≤ Python, –ø–æ—è–≤–∏–≤—à–∏–π—Å—è –≤ –≤–µ—Ä—Å–∏–∏ 3.6. –û–Ω–∏ –æ–±–µ—Å–ø–µ—á–∏–≤–∞—é—Ç –ø—Ä–æ—Å—Ç–æ–π, —á–∏—Ç–∞–µ–º—ã–π –∏ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω—ã–π —Å–ø–æ—Å–æ–± —Å–æ–∑–¥–∞–Ω–∏—è —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å—Ç—Ä–æ–∫.

##### üî∏ –ë–∞–∑–æ–≤—ã–π —Å–∏–Ω—Ç–∞–∫—Å–∏—Å
```python
# –û—Å–Ω–æ–≤–Ω–æ–π —Å–∏–Ω—Ç–∞–∫—Å–∏—Å f-—Å—Ç—Ä–æ–∫

# f-—Å—Ç—Ä–æ–∫–∞ –Ω–∞—á–∏–Ω–∞–µ—Ç—Å—è —Å –ø—Ä–µ—Ñ–∏–∫—Å–∞ f –∏–ª–∏ F

print(message)  # –ü—Ä–∏–≤–µ—Ç, –º–µ–Ω—è –∑–æ–≤—É—Ç Alice –∏ –º–Ω–µ 25 –ª–µ—Ç

# –ú–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –∑–∞–≥–ª–∞–≤–Ω—É—é F
message_caps = F"–ü—Ä–∏–≤–µ—Ç, {name}!"

# –°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Å –¥—Ä—É–≥–∏–º–∏ —Å–ø–æ—Å–æ–±–∞–º–∏ —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è

# f-—Å—Ç—Ä–æ–∫–∞ (—Å–∞–º—ã–π —á–∏—Ç–∞–µ–º—ã–π –∏ –±—ã—Å—Ç—Ä—ã–π)

print("–ú–µ—Ç–æ–¥ format:", format_way)
print("F-—Å—Ç—Ä–æ–∫–∞:", f_string_way)
```

##### üî∏ –§–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —á–∏—Å–µ–ª —Å –ø–ª–∞–≤–∞—é—â–µ–π —Ç–æ—á–∫–æ–π

```python
from datetime import datetime, date, timedelta
import locale

# –ü–æ–ø—ã—Ç–∞–µ–º—Å—è —É—Å—Ç–∞–Ω–æ–≤–∏—Ç—å —Ä—É—Å—Å–∫—É—é –ª–æ–∫–∞–ª—å (–º–æ–∂–µ—Ç –Ω–µ —Ä–∞–±–æ—Ç–∞—Ç—å –Ω–∞ –≤—Å–µ—Ö —Å–∏—Å—Ç–µ–º–∞—Ö)

formatted_datetime = f"–î–∞—Ç–∞ –∏ –≤—Ä–µ–º—è: {now:%Y-%m-%d %H:%M:%S}"

# –†–∞–∑–ª–∏—á–Ω—ã–µ —Ñ–æ—Ä–º–∞—Ç—ã –¥–∞—Ç—ã
date_formats = [
    f"ISO —Ñ–æ—Ä–º–∞—Ç: {now:%Y-%m-%d}",
    f"–ï–≤—Ä–æ–ø–µ–π—Å–∫–∏–π: {now:%d.%m.%Y}",

def humanize_timedelta(td):
    total_seconds = int(td.total_seconds())
    days = total_seconds // 86400
    hours = (total_seconds % 86400) // 3600
    minutes = (total_seconds % 3600) // 60
    seconds = total_seconds % 60
    
    parts = []
    if days:
        parts.append(f"{days} –¥–Ω.")
    if hours:
        parts.append(f"{hours} —á.")
    if minutes:
        parts.append(f"{minutes} –º–∏–Ω.")
    if seconds or not parts:
        parts.append(f"{seconds} —Å–µ–∫.")
    
    return " ".join(parts)

    timedelta(hours=2, minutes=15),
    timedelta(days=1, hours=3, minutes=45),
    timedelta(weeks=2, days=3),
]

for period in time_periods:
    print(f"–ü–µ—Ä–∏–æ–¥ {period} = {humanize_timedelta(period)}")
```
