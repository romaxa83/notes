def binary_search(arr, target):
    low = 0
    high = len(arr) - 1

    while low <= high:
        mid = (low + high) // 2  # –ù–∞—Ö–æ–¥–∏–º —Å–µ—Ä–µ–¥–∏–Ω—É
        guess = arr[mid]
        
        if guess == target:
            return mid  # –ù–∞—à–ª–∏! –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∏–Ω–¥–µ–∫—Å
        if guess > target:
            high = mid - 1  # –°–ª–∏—à–∫–æ–º –º–Ω–æ–≥–æ, –æ—Ç—Å–µ–∫–∞–µ–º –ø—Ä–∞–≤—É—é —á–∞—Å—Ç—å
        else:

    return -1  # –≠–ª–µ–º–µ–Ω—Ç–∞ –Ω–µ—Ç –≤ —Å–ø–∏—Å–∫–µ

# –¢–µ—Å—Ç (–º–∞—Å—Å–∏–≤ –û–ë–Ø–ó–ê–¢–ï–õ–¨–ù–û –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω)
