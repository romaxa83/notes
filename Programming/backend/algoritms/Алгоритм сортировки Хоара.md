def quick_sort(a):

		x = a[random.randint(0, len(a)-1)] # —Å–ª—É—á–∞–π–Ω–æ–µ –ø–æ—Ä–æ–≥–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ (–¥–ª—è —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è –Ω–∞ –º–∞–ª—ã–µ –∏ –±–æ–ª—å—à–∏–µ)

		eq = [u for u in a if u == x]
		hig = [u for u in a if u > x]
		a = quick_sort(low) + eq + quick_sort(hig)
