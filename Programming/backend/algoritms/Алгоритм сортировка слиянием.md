def merge_list(a, b):
    c = []
    N = len(a)
    M = len(b)

    i = 0
    j = 0
    while i < N and j < M:
        if a[i] <= b[j]:
            c.append(a[i])
            i += 1
        else:
            c.append(b[j])
            j += 1

    c += a[i:] + b[j:]
    return c

# —Ñ—É–Ω–∫—Ü–∏—è –¥–µ–ª–µ–Ω–∏—è —Å–ø–∏—Å–∫–∞ –∏ —Å–ª–∏—è–Ω–∏—è —Å–ø–∏—Å–∫–æ–≤ –≤ –æ–±—â–∏–π –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫
def split_and_merge_list(a):

        a2 = split_and_merge_list(a2)

    return merge_list(a1, a2)   # —Å–ª–∏—è–Ω–∏–µ –¥–≤—É—Ö –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö —Å–ø–∏—Å–∫–æ–≤ –≤ –æ–¥–∏–Ω
