–û—Å–Ω–æ–≤–Ω–∞—è —Ä–∞–±–æ—Ç–∞ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏—è–º–∏  `sync.Lock()` –∏ `sync.Unlock()`, –ø–µ—Ä–≤–∞—è –∏–∑  –∫–æ—Ç–æ—Ä—ã—Ö –±–ª–æ–∫–∏—Ä—É–µ—Ç, –∞ –≤—Ç–æ—Ä–∞—è —Ä–∞–∑–±–ª–æ–∫–∏—Ä—É–µ—Ç –º—å—é—Ç–µ–∫—Å `sync.Mutex`. –ö–æ–≥–¥–∞ –º—å—é—Ç–µ–∫—Å  –±–ª–æ–∫–∏—Ä–æ–≤–∞–Ω, —ç—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ –Ω–∏–∫—Ç–æ –¥—Ä—É–≥–æ–π  –Ω–µ –º–æ–∂–µ—Ç –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞—Ç—å —ç—Ç–æ—Ç –º—å—é—Ç–µ–∫—Å,  –ø–æ–∫–∞ –æ–Ω –Ω–µ –±—É–¥–µ—Ç –æ—Å–≤–æ–±–æ–∂–¥–µ–Ω —Å –ø–æ–º–æ—â—å—é  —Ñ—É–Ω–∫—Ü–∏–∏ `sync.Unlock()`

==–ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è==
```go
package main

import (
	"fmt"
	"os"
	"strconv"
	"sync"
	"time"
)

var (
	m sync.Mutex
	v1 int
)

func change(i int) {
	m.Lock()
	time.Sleep(time.Second)
	v1 = v1 + 1
	if v1%10 == 0 {
		v1 = v1 - 10*i
	}
	m.Unlock()
}

func read() int {
	m.Lock()
	a := v1
	m.Unlock()
	return a
}

func main() {
	if len(os.Args) != 2 {
		fmt.Println("Please give me an integer!")
		return
	}

	numGR, err := strconv.Atoi(os.Args[1])
	if err != nil {
		fmt.Println(err)
		return
	}

	var waitGroup sync.WaitGroup
	fmt.Printf("%d ", read())
	for i := 0; i < numGR; i++ {
		waitGroup.Add(1)
		go func(i int) {
			defer waitGroup.Done()
			change(i)
			fmt.Printf("-> %d", read())
		}(i)
	}
	waitGroup.Wait()
	fmt.Printf("-> %d\n", read())
}
```

–≠—Ç–æ —É–ª—É—á—à–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è `sync.Mutex`.

–¢–æ–ª—å–∫–æ –æ–¥–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ —Ä–∞–∑—Ä–µ—à–∞–µ—Ç—Å—è –≤—ã–ø–æ–ª–Ω—è—Ç—å –æ–ø–µ—Ä–∞—Ü–∏–∏ –∑–∞–ø–∏—Å–∏ —Å –º—å—é—Ç–µ–∫—Å–æ–º `sync.RWMutex`,  —É –Ω–∞—Å –º–æ–∂–µ—Ç –±—ã—Ç—å –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ñ—É–Ω–∫—Ü–∏–π, –≤–ª–∞–¥–µ—é—â–∏—Ö –º—å—é—Ç–µ–∫—Å–æ–º `sync.RWMutex` –∏ –≤—ã–ø–æ–ª–Ω—è—é—â–∏—Ö —á—Ç–µ–Ω–∏–µ. –û–¥–Ω–∞–∫–æ —Å–ª–µ–¥—É–µ—Ç —É—á–∏—Ç—ã–≤–∞—Ç—å, —á—Ç–æ, –ø–æ–∫–∞ –≤—Å–µ —Ñ—É–Ω–∫—Ü–∏–∏, —á–∏—Ç–∞—é—â–∏–µ –º—å—é—Ç–µ–∫—Å —Ç–∏–ø–∞ `sync.RWMutex`, –Ω–µ —Ä–∞–∑–±–ª–æ–∫–∏—Ä—É—é—Ç —ç—Ç–æ—Ç –º—å—é—Ç–µ–∫—Å, –≤—ã –Ω–µ —Å–º–æ–∂–µ—Ç–µ –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞—Ç—å –µ–≥–æ –¥–ª—è –∑–∞–ø–∏—Å–∏.

–†–∞–±–æ—Ç–∞—Ç—å —Å –º—å—é—Ç–µ–∫—Å–æ–º `sync.RWMutex` –≤–∞–º –ø–æ–º–æ–≥—É—Ç —Ñ—É–Ω–∫—Ü–∏–∏ `RLock()` –∏ `RUnlock()`, –∫–æ—Ç–æ—Ä—ã–µ –±–ª–æ–∫–∏—Ä—É—é—Ç  –∏ —Ä–∞–∑–±–ª–æ–∫–∏—Ä—É—é—Ç –º—å—é—Ç–µ–∫—Å –¥–ª—è —á—Ç–µ–Ω–∏—è —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ —Ñ—É–Ω–∫—Ü–∏–∏ `Lock()` –∏ `Unlock()`, –∏—Å–ø–æ–ª—å–∑—É–µ–º—ã–µ –≤ –º—å—é—Ç–µ–∫—Å–µ `sync.Mutex`, –ø–æ-–ø—Ä–µ–∂–Ω–µ–º—É –ø—Ä–∏–º–µ–Ω—è—é—Ç—Å—è –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –∑–∞–±–ª–æ–∫–∏—Ä–æ–≤–∞—Ç—å –∏ —Ä–∞–∑–±–ª–æ–∫–∏—Ä–æ–≤–∞—Ç—å –º—å—é—Ç–µ–∫—Å sync.RWMutex –¥–ª—è –∑–∞–ø–∏—Å–∏.

==–ü–†–ò–ú–ï–† (—Å–∫—Ä–∏–ø—Ç —Å–æ–¥–µ—Ä–∂–∏—Ç –¥–≤–µ –æ—Ç–ª–∏—á–∞—é—â–∏–µ—Å—è –≤–µ—Ä—Å–∏–∏ –æ–¥–Ω–æ–π –∏ —Ç–æ–π –∂–µ —Ñ—É–Ω–∫—Ü–∏–∏. –í –ø–µ—Ä–≤–æ–π –∏–∑ —ç—Ç–∏—Ö —Ñ—É–Ω–∫—Ü–∏–π –¥–ª—è —á—Ç–µ–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –º—å—é—Ç–µ–∫—Å —Ç–∏–ø–∞ sync.RWMutex ,–∞ –≤—Ç–æ—Ä–æ–π ‚Äî –º—å—é—Ç–µ–∫—Å —Ç–∏–ø–∞ sync.Mutex.==

import (
	"fmt"
	"os"
	"sync"
	"time"
)

var Password = secret{password: "myPassword"}
type secret struct {
	RWM      sync.RWMutex
	M        sync.Mutex
	password string
}

// —Ñ—É–Ω–∫—Ü–∏—è –∏–∑–º–µ–Ω—è–µ—Ç –æ–±—â—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é.

// –¥–ª—è —á–µ–≥–æ –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏–∏ Lock() –∏ Unlock()
func Change(c *secret, pass string) {
	c.RWM.Lock()
	fmt.Println("LChange")
	time.Sleep(10 * time.Second)
	c.password = pass
	c.RWM.Unlock()
}

// —Ñ—É–Ω–∫—Ü–∏–∏ –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è RLock() –∏ RUnlock(),
// –ø–æ—Å–∫–æ–ª—å–∫—É –µ–µ –∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏–π —Ä–∞–∑–¥–µ–ª –Ω—É–∂–µ–Ω 
// –¥–ª—è —á—Ç–µ–Ω–∏—è –æ–±—â–µ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
// –Ω–µ—Å–º–æ—Ç—Ä—è –Ω–∞ —Ç–æ, —á—Ç–æ –Ω–µ—Å–∫–æ–ª—å–∫–æ –≥–æ—Ä—É—Ç–∏–Ω –º–æ–≥—É—Ç 
// —á–∏—Ç–∞—Ç—å –æ–±—â—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é, —ç—Ç—É –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é –Ω–µ–ª—å–∑—è
// –∏–∑–º–µ–Ω–∏—Ç—å –±–µ–∑ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–π Lock() –∏ Unlock()
func show(c *secret) string {
	c.RWM.RLock()
	fmt.Print("show")
	time.Sleep(3 * time.Second)
	defer c.RWM.RUnlock()
	return c.password
}

// —Ä–∞–∑–ª–∏—á–∏–µ –º–µ–∂–¥—É –∫–æ–¥–æ–º —Ñ—É–Ω–∫—Ü–∏–π showWithLock()

// –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è –º–æ–Ω–æ–ø–æ–ª—å–Ω–∞—è –±–ª–æ–∫–∏—Ä–æ–≤–∫–∞ –¥–ª—è —á—Ç–µ–Ω–∏—è
// –¥–∞–Ω–Ω—ã—Ö
func showWithLock(c *secret) string {
	c.RWM.Lock()
	fmt.Println("showWithLock")
	time.Sleep(3 * time.Second)
	defer c.RWM.Unlock()
	return c.password
}

func main(){
	var showFunction = func(c *secret) string { return "" }

	if len(os.Args) != 2 {
		fmt.Println("Using sync.RWMutex!")
		showFunction = show
	} else {
		fmt.Println("Using sync.Mutex!")
		showFunction = showWithLock
	}
	var waitGroup sync.WaitGroup
	fmt.Println("Pass:", showFunction(&Password))

	for i := 0; i < 15; i++ {
		waitGroup.Add(1)
		go func() {
			defer waitGroup.Done()
			fmt.Println("Go Pass:", showFunction(&Password))
		}()	
	}
	go func() {
		waitGroup.Add(1)
		defer waitGroup.Done()
		Change(&Password, "123456")
	}()	

	waitGroup.Wait()
	fmt.Println("Pass:", showFunction(&Password))
}
