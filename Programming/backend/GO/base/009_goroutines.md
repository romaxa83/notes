==–ü–†–ò–ú–ï–† (—Å–∫—Ä–∏–ø—Ç —Ä–µ—à–∞–µ—Ç –ø—Ä–æ—Å—Ç—É—é –∑–∞–¥–∞—á—É: –æ–Ω –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç —Ü–µ–ª—ã–µ —á–∏—Å–ª–∞ –∏ –≤—ã–≤–æ–¥–∏—Ç –∏—Ö –∫–≤–∞–¥—Ä–∞—Ç—ã –ø–æ—Å—Ä–µ–¥—Å—Ç–≤–æ–º –æ—Ç–¥–µ–ª—å–Ω–æ–π –≥–æ—Ä—É—Ç–∏–Ω—ã –¥–ª—è –æ–±—Å–ª—É–∂–∏–≤–∞–Ω–∏—è –∫–∞–∂–¥–æ–≥–æ –∑–∞–ø—Ä–æ—Å–∞)==

	"fmt"
	"os"
	"strconv"
	"sync"
	"time"
)

// –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä–∞ –∫–∞–∂–¥–æ–º—É –∑–∞–ø—Ä–æ—Å—É (—Å–æ–¥–µ—Ä–∂–∏—Ç –≤—Ö–æ–¥–Ω—ã–µ

type Client struct {
	id      int
	integer int
}

// –°—Ç—Ä—É–∫—Ç—É—Ä–∞ Data –Ω—É–∂–Ω–∞ –¥–ª—è –≥—Ä—É–ø–ø–∏—Ä–æ–≤–∫–∏ –¥–∞–Ω–Ω—ã—Ö
// Client —Å —Ä–µ–∞–ª—å–Ω—ã–º–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞–º–∏ (—Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –∑–∞–ø—Ä–æ—Å–∞)
type Data struct {
	job    Client
	square int
}

// –ë—É—Ñ–µ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã–µ –∫–∞–Ω–∞–ª—ã clients –∏ data –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –¥–ª—è
// –ø–æ–ª—É—á–µ–Ω–∏—è –Ω–æ–≤—ã—Ö –∫–ª–∏–µ–Ω—Ç—Å–∫–∏—Ö –∑–∞–ø—Ä–æ—Å–æ–≤ –∏ –∑–∞–ø–∏—Å–∏ 
// —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ. –ï—Å–ª–∏ –≤—ã —Ö–æ—Ç–∏—Ç–µ, —á—Ç–æ–±—ã
// –ø—Ä–æ–≥—Ä–∞–º–º–∞ —Ä–∞–±–æ—Ç–∞–ª–∞ –±—ã—Å—Ç—Ä–µ–µ, –º–æ–∂–µ—Ç–µ —É–≤–µ–ª–∏—á–∏—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ
// –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ size
var (
	size = 10
	clients = make(chan Client, size)
	data = make(chan Data, size)
)

// –º–µ—Ç–æ–¥ —á–∏—Ç–∞–µ—Ç –∫–∞–Ω–∞–ª clients –∏ –ø–æ–ª—É—á–∞–µ—Ç –æ—Ç—Ç—É–¥–∞
// –Ω–æ–≤—ã–µ –∑–∞–ø—Ä–æ—Å—ã –Ω–∞ –æ–±—Å–ª—É–∂–∏–≤–∞–Ω–∏–µ. –ü–æ—Å–ª–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –æ–±—Ä–∞–±–æ—Ç–∫–∏
// —Ä–µ–∑—É–ª—å—Ç–∞—Ç –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –≤ –∫–∞–Ω–∞–ª data. –ó–∞–¥–µ—Ä–∂–∫–∞,
// –∫–æ—Ç–æ—Ä–∞—è –≤–≤–æ–¥–∏—Ç—Å—è —Å –ø–æ–º–æ—â—å—é –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ time.Sleep(time.Second),
// –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ–π, –Ω–æ –¥–∞–µ—Ç –ª—É—á—à–µ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ
// –æ —Ç–æ–º, –∫–∞–∫ –±—É–¥—É—Ç –≤—ã–≤–æ–¥–∏—Ç—å—Å—è –Ω–∞ —ç–∫—Ä–∞–Ω —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–æ–≥—Ä–∞–º–º—ã.
// –ù–∞–∫–æ–Ω–µ—Ü, –Ω–µ –∑–∞–±—É–¥—å—Ç–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –≤ —Ñ—É–Ω–∫—Ü–∏–∏ worker()
// —É–∫–∞–∑–∞—Ç–µ–ª—å –¥–ª—è –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ sync.WaitGroup, –∏–Ω–∞—á–µ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è
// sync.WaitGroup —Å–∫–æ–ø–∏—Ä—É–µ—Ç—Å—è –∏ –±—É–¥–µ—Ç –±–µ—Å–ø–æ–ª–µ–∑–Ω–æ–π
func worker(w *sync.WaitGroup) {
	for c := range clients {
		square := c.integer * c.integer
		output := Data{c, square}
		data <- output
		time.Sleep(time.Second)
	}

	w.Done()
}

// –ù–∞–∑–Ω–∞—á–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ makeWP() —Å–æ—Å—Ç–æ–∏—Ç –≤ —Ç–æ–º, —á—Ç–æ–±—ã
// —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –≥–æ—Ä—É—Ç–∏–Ω
// worker() –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –≤—Å–µ—Ö –∑–∞–ø—Ä–æ—Å–æ–≤. –§—É–Ω–∫—Ü–∏—è w.Add(1)
// –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –∏–∑ makeWP(), –æ–¥–Ω–∞–∫–æ w.Done() –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è
// –∏–∑ worker() –ø–æ—Å–ª–µ —Ç–æ–≥–æ, –∫–∞–∫ –æ–±—Ä–∞–±–æ—Ç—á–∏–∫ –∑–∞–≤–µ—Ä—à–∏—Ç —Å–≤–æ—é
// —Ä–∞–±–æ—Ç—É
func makeWP(n int) {
	var w sync.WaitGroup
	for i := 0; i < n; i++ {
		w.Add(1)
		go worker(&w)
	}
	w.Wait()
	close(data)
}

// –ù–∞–∑–Ω–∞—á–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ create() ‚Äî –ø—Ä–∞–≤–∏–ª—å–Ω–æ —Å–æ–∑–¥–∞—Ç—å
// –≤—Å–µ –∑–∞–ø—Ä–æ—Å—ã, –∏—Å–ø–æ–ª—å–∑—É—è —Ç–∏–ø Client, –∞ –∑–∞—Ç–µ–º –∑–∞–ø–∏—Å–∞—Ç—å
// –∏—Ö –≤ –∫–∞–Ω–∞–ª clients –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏. –û–±—Ä–∞—Ç–∏—Ç–µ –≤–Ω–∏–º–∞–Ω–∏–µ,
// —á—Ç–æ –∫–∞–Ω–∞–ª clients —á–∏—Ç–∞–µ—Ç—Å—è —Ñ—É–Ω–∫—Ü–∏–µ–π worker()
func create(n int) {
	for i := 0; i < n; i++ {
		c := Client{i, i}
		clients <- c
	}
	close(clients)
}

func main() {
	fmt.Println("Capacity of clients:", cap(clients))
	fmt.Println("Capacity of data:", cap(data))
	if len(os.Args) != 3 {
		fmt.Println("Need #jobs and #workers!")
		os.Exit(1)
	}
	nJobs, err := strconv.Atoi(os.Args[1])
	if err != nil {
		fmt.Println(err)
		return
	}
	nWorkers, err := strconv.Atoi(os.Args[2])
	if err != nil {
		fmt.Println(err)
		return
	}

// –°–Ω–∞—á–∞–ª–∞ –º—ã –≤—ã–∑—ã–≤–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é create(), —á—Ç–æ–±—ã –∏–º–∏—Ç–∏—Ä–æ–≤–∞—Ç—å
// –∫–ª–∏–µ–Ω—Ç—Å–∫–∏–µ –∑–∞–ø—Ä–æ—Å—ã, –∫–æ—Ç–æ—Ä—ã–µ —Å–ª–µ–¥—É–µ—Ç –æ–±—Ä–∞–±–æ—Ç–∞—Ç—å.
// –î–ª—è —á—Ç–µ–Ω–∏—è –∫–∞–Ω–∞–ª–∞ data –∏ –≤—ã–≤–æ–¥–∞ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
// –Ω–∞ —ç–∫—Ä–∞–Ω –ø—Ä–∏–º–µ–Ω—è–µ—Ç—Å—è –∞–Ω–æ–Ω–∏–º–Ω–∞—è –≥–æ—Ä—É—Ç–∏–Ω–∞.
// –ö–∞–Ω–∞–ª finished –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –±–ª–æ–∫–∏—Ä–æ–≤–∫–∏ –ø—Ä–æ–≥—Ä–∞–º–º—ã –¥–æ

// –∫–∞–Ω–∞–ª–∞ data. –ü–æ—ç—Ç–æ–º—É –∫–∞–Ω–∞–ª finished –Ω–µ –Ω—É–∂–¥–∞–µ—Ç—Å—è –≤ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–º —Ç–∏–ø–µ.
// –ó–∞—Ç–µ–º –º—ã –≤—ã–∑—ã–≤–∞–µ–º —Ñ—É–Ω–∫—Ü–∏—é makeWP(), –∫–æ—Ç–æ—Ä–∞—è –∏ –≤—ã–ø–æ–ª–Ω—è–µ—Ç
// –æ–±—Ä–∞–±–æ—Ç–∫—É –∑–∞–ø—Ä–æ—Å–æ–≤. –û–ø–µ—Ä–∞—Ç–æ—Ä <-finished –≤ –±–ª–æ–∫–∞—Ö fmt.Printf()
